SELECT DISTINCT ?uri ?name SAMPLE(?info) AS ?info ?thumbnail SAMPLE(?date) AS ?date COUNT(DISTINCT ?type) AS ?indegree WHERE{    ?uri dbo:birthDate ?date	FILTER (( datatype(?date) = xsd:date ) ||		( datatype(?date) = xsd:dateTime ))	FILTER (regex(str(?date), datum) ).    ?uri rdf:type ?isKindOf	FILTER((regex(str(?isKindOf),"ontology/BeautyQueen"))||	(regex(str(?isKindOf),"ontology/Celebrity"))||	(regex(str(?isKindOf),"ontology/Model"))||	(regex(str(?isKindOf),"ontology/Artist"))||	(regex(str(?isKindOf),"ontology/MovieDirector"))||	(regex(str(?isKindOf),"ontology/Presenter"))||	(regex(str(?isKindOf),"ontology/Producer"))||	(regex(str(?isKindOf),"ontology/TelevisionDirector"))||	(regex(str(?isKindOf),"ontology/TelivisionPersonality"))||	(regex(str(?isKindOf),"ontology/TheatreDirector"))||	(regex(str(?isKindOf),"ontology/Writer"))).    ?uri rdfs:label ?name	FILTER(langMatches(lang(?name), "EN")).    ?type ?property ?uri.    OPTIONAL {?uri dbo:thumbnail ?thumbnail }.    ?uri dbp:shortDescription ?info.} ORDER BY DESC(?indegree)
