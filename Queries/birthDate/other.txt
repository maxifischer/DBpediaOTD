SELECT DISTINCT ?uri ?name SAMPLE(?info) AS ?info ?thumbnail SAMPLE(?date) AS ?date COUNT(DISTINCT ?type) AS ?indegree WHERE{    ?uri dbo:birthDate ?date	FILTER (( datatype(?date) = xsd:date ) ||		( datatype(?date) = xsd:dateTime ))	FILTER (regex(str(?date), datum) ).    ?uri rdf:type ?isKindOf	FILTER( (regex(str(?isKindOf),"ontology/PlayboyPlaymate"))||		(regex(str(?isKindOf),"ontology/Chef"))||		(regex(str(?isKindOf),"ontology/Farmer"))||		(regex(str(?isKindOf),"ontology/Orphan"))||		(regex(str(?isKindOf),"ontology/Cleric"))||		(regex(str(?isKindOf),"ontology/Religious"))||		(regex(str(?isKindOf),"ontology/Criminal"))||		(regex(str(?isKindOf),"ontology/Judge"))||		(regex(str(?isKindOf),"ontology/Lawyer"))||		(regex(str(?isKindOf),"ontology/OrganisationMember"))).    ?uri rdfs:label ?name	FILTER(langMatches(lang(?name), "EN")).    ?type ?property ?uri.    OPTIONAL {?uri dbo:thumbnail ?thumbnail }.    ?uri dbp:shortDescription ?info.} ORDER BY DESC(?indegree)
