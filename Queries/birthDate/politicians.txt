SELECT DISTINCT ?uri ?name SAMPLE(?info) AS ?info ?thumbnail SAMPLE(?date) AS ?date COUNT(DISTINCT ?type) AS ?indegree WHERE{    ?uri dbo:birthDate ?date	FILTER (( datatype(?date) = xsd:date ) ||		( datatype(?date) = xsd:dateTime ))	FILTER (regex(str(?date), datum) ).?uri rdf:type ?isKindOf	FILTER( (regex(str(?isKindOf),"ontology/Journalist"))||		(regex(str(?isKindOf),"ontology/Ambassador"))||		(regex(str(?isKindOf),"ontology/Aristocrat"))||		(regex(str(?isKindOf),"ontology/Monarch"))||		(regex(str(?isKindOf),"ontology/Noble"))||		(regex(str(?isKindOf),"ontology/RomanEmperor"))||		(regex(str(?isKindOf),"ontology/Royality"))||		(regex(str(?isKindOf),"ontology/MemberResistanceMovement"))||		(regex(str(?isKindOf),"ontology/MilitaryPerson"))||		(regex(str(?isKindOf),"ontology/OfficeHolder"))).    ?uri rdfs:label ?name	FILTER(langMatches(lang(?name), "EN")).    ?type ?property ?uri.    OPTIONAL {?uri dbo:thumbnail ?thumbnail }.    ?uri dbp:shortDescription ?info.} ORDER BY DESC(?indegree)
