SELECT DISTINCT ?uri ?name SAMPLE(?info) AS ?info ?thumbnail SAMPLE(?date) AS ?date COUNT(DISTINCT ?type) AS ?indegree WHERE{    ?uri dbo:deathDate ?date	FILTER (( datatype(?date) = xsd:date ) ||		( datatype(?date) = xsd:dateTime ))	FILTER (regex(str(?date), datum) ).    ?uri rdf:type ?isKindOf	FILTER( (regex(str(?isKindOf),"ontology/Athlete"))||		(regex(str(?isKindOf),"ontology/Coach"))||		(regex(str(?isKindOf),"ontology/HorseTrainer"))||		(regex(str(?isKindOf),"ontology/Referee"))||		(regex(str(?isKindOf),"ontology/SportManager"))).    ?uri rdfs:label ?name	FILTER(langMatches(lang(?name), "EN")).    ?type ?property ?uri.    OPTIONAL {?uri dbo:thumbnail ?thumbnail }.    ?uri dbp:shortDescription ?info.} ORDER BY DESC(?indegree)
