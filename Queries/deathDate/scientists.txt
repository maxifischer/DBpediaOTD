SELECT DISTINCT ?uri ?name SAMPLE(?info) AS ?info ?thumbnail SAMPLE(?date) AS ?date COUNT(DISTINCT ?type) AS ?indegree WHERE{    ?uri dbo:deathDate ?date	FILTER (( datatype(?date) = xsd:date ) ||		( datatype(?date) = xsd:dateTime ))	FILTER (regex(str(?date), datum) ).    ?uri rdf:type ?isKindOf	FILTER( (regex(str(?isKindOf),"ontology/Astronaut"))||		(regex(str(?isKindOf),"ontology/Architect"))||		(regex(str(?isKindOf),"ontology/Archeologist"))||		(regex(str(?isKindOf),"ontology/Egyptologist"))||		(regex(str(?isKindOf),"ontology/Economist"))||		(regex(str(?isKindOf),"ontology/Engineer"))||		(regex(str(?isKindOf),"ontology/Linguist"))||		(regex(str(?isKindOf),"ontology/Scientist"))||		(regex(str(?isKindOf),"ontology/Philosopher"))||		(regex(str(?isKindOf),"ontology/Psychologist"))).    ?uri rdfs:label ?name	FILTER(langMatches(lang(?name), "EN")).    ?type ?property ?uri.    OPTIONAL {?uri dbo:thumbnail ?thumbnail }.    ?uri dbp:shortDescription ?info.} ORDER BY DESC(?indegree)
